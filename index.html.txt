    <div class="container" id="ilan-konteyner">
        <p style='text-align:center;'>Şu an güncel ilanlar yükleniyor...</p>
    </div>

    <script>
        // Python'un oluşturacağı 'ilanlar.json' dosyasını arayıp bulma komutu
        fetch('ilanlar.json')
            .then(cevap => cevap.json())
            .then(veri => {
                ilanlariGoster(veri); // Veri gelince ekranı çizmeye başla
            })
            .catch(hata => {
                console.error("Hata:", hata);
                document.getElementById('ilan-konteyner').innerHTML = 
                    "<p style='text-align:center;'>Henüz yeni ilan bulunamadı veya taranıyor...</p>";
            });

        // Gelen gerçek verileri kutucuklara yerleştirme işlemi
        function ilanlariGoster(ilanListesi) {
            const konteyner = document.getElementById('ilan-konteyner');
            konteyner.innerHTML = ''; 
            
            // Tarihe göre sırala (En yeni en üstte)
            ilanListesi.sort((a, b) => new Date(b.tarih) - new Date(a.tarih));

            ilanListesi.forEach(ilan => {
                const kart = `
                    <div class="ilan-karti">
                        <h2 class="ilan-baslik">${ilan.baslik}</h2>
                        <div class="sirket-isim">🏢 ${ilan.sirket} - 📍 ${ilan.konum}</div>
                        <div class="detaylar">
                            <span class="etiket">Güncel İlan</span>
                            <span class="tarih">🕒 ${ilan.tarih}</span>
                        </div>
                        <a href="${ilan.link}" target="_blank" class="link-buton">İlan Detayına Git</a>
                    </div>
                `;
                konteyner.innerHTML += kart;
            });

            const simdi = new Date();
            document.getElementById('son-guncelleme').innerText = 
                "Son Güncelleme: " + simdi.toLocaleTimeString('tr-TR', {hour: '2-digit', minute:'2-digit'});
        }
    </script>
